#include <stdio.h>
#include <stdlib.h>
#include <time.h>

// Function to check if page is in frame
int isHit(int frames[], int f, int page) {
    for (int i = 0; i < f; i++) {
        if (frames[i] == page)
            return 1;
    }

    return 0;
}

// ---------------- FIFO Page Replacement ----------------
int fifo(int pages[], int n, int f) {
    int frames[10];
    int faults = 0;
    int next = 0;

    // Initialize frames
    for (int i = 0; i < f; i++)
        frames[i] = -1;

    // Process pages
    for (int i = 0; i < n; i++) {
        if (!isHit(frames, f, pages[i])) {
            frames[next] = pages[i];
            next = (next + 1) % f;
            faults++;
        }
    }

    return faults;
}

// ---------------- MAIN ----------------
int main(int argc, char *argv[]) {
    if (argc != 3) {
        printf("Usage: %s <length> <frames>\n", argv[0]);
        return 1;
    }

    int n = atoi(argv[1]); // Length of reference string
    int f = atoi(argv[2]); // Number of frames

    int pages[100];

    // Generate random pages (0–9)
    srand(time(0));

    printf("Page Reference String:\n");

    for (int i = 0; i < n; i++) {
        pages[i] = rand() % 10;
        printf("%d ", pages[i]);
    }

    printf("\n\n");

    // Run FIFO algorithm
    int fifo_faults = fifo(pages, n, f);

    // Output
    printf("Frames = %d\n", f);
    printf("FIFO Page Faults = %d\n", fifo_faults);

    return 0;
}

gcc fifo.c -o fifo
